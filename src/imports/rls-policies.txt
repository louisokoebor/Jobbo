-- =============================================
-- 1. public.users
-- =============================================
alter table public.users enable row level security;

drop policy if exists "users: read own" on public.users;
drop policy if exists "users: update own safe fields" on public.users;

create policy "users: read own"
on public.users for select
to authenticated
using ((select auth.uid()) = id);

create policy "users: update own safe fields"
on public.users for update
to authenticated
using ((select auth.uid()) = id)
with check ((select auth.uid()) = id);

-- =============================================
-- 4. public.generated_cvs
-- =============================================
alter table public.generated_cvs enable row level security;

drop policy if exists "generated_cvs: read own" on public.generated_cvs;
drop policy if exists "generated_cvs: insert own" on public.generated_cvs;
drop policy if exists "generated_cvs: update own" on public.generated_cvs;
drop policy if exists "generated_cvs: delete own" on public.generated_cvs;

create policy "generated_cvs: read own"
on public.generated_cvs for select
to authenticated
using (
  exists (
    select 1 from public.applications a
    where a.id = generated_cvs.application_id
      and a.user_id = (select auth.uid())
  )
);

create policy "generated_cvs: insert own"
on public.generated_cvs for insert
to authenticated

  exists (
    select 1 from public.applications a
